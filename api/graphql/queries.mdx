## Object Queries

### Get Object Details

Query an object with all available fields:

```graphql
query GetObjectDetails($id: SuiAddress!) {
  object(address: $id) {
    address
    version
    digest
    storageRebate
    
    owner {
      __typename
      ... on AddressOwner {
        owner {
          address
        }
      }
      ... on Shared {
        initialSharedVersion
      }
      ... on Parent {
        parent {
          address
        }
      }
      ... on Immutable {
        __typename
      }
    }
    
    previousTransactionBlock {
      digest
    }
    
    asMoveObject {
      hasPublicTransfer
      contents {
        type {
          repr
          layout
          signature
        }
        json
        bcs
      }
    }
    
    display {
      key
      value
      error
    }
  }
}
```

### Get Dynamic Fields

```graphql

            }

              repr
            }

            bcs
          }
        }
      }
      pageInfo {
        hasNextPage
        endCursor
```
```graphql
      }
    }
  }
}
```

### Query Transaction Blocks

```graphql
query QueryTransactionBlocks(
  $filter: TransactionBlockFilter
  $first: Int
  $after: String
) {
  transactionBlocks(filter: $filter, first: $first, after: $after) {
    nodes {
      digest
      sender {
        address
      }
      effects {
        status
        timestamp
        gasEffects {
          gasSummary {
            computationCost
            storageCost
          }
        }
      }
    }
    pageInfo {
      hasNextPage
      endCursor
    }
  }
}
```

Filter by sender:
```json
{
  "filter": {
    "signAddress": "0x..."
  },
  "first": 50
}
```

Filter by recipient:
```json
{
  "filter": {
    "recvAddress": "0x..."
  },
  "first": 50
}
```

```graphql
      json
      bcs
      
      transactionBlock {
        digest
      }
    }

      hasNextPage
      endCursor
    }
  }

```

Filter by event type:
```json
{
  "filter": {
    "eventType": "0x2::coin::CoinCreated<0x2::sui::SUI>"
  },
  "first": 20
}
```

Filter by sender:
```json
{
  "filter": {
    "sender": "0x..."
  },
  "first": 20
}
```

Filter by package:
```json
{
  "filter": {
    "emittingPackage": "0x2"
  },
  "first": 20
}
```

## Combining Queries

### Get Complete Address Profile

```graphql
query GetAddressProfile($address: SuiAddress!) {
  address(address: $address) {
    address
    
    # Balances
    balances(first: 100) {
      nodes {
        coinType {
          repr
        }
        totalBalance
      }
    }
    
    # Recent transactions
    transactionBlocks(first: 10) {
      nodes {
        digest
        effects {
          timestamp
          status
        }
      }
    }
    
    # Owned objects
    objects(first: 20) {
      nodes {
        address
        asMoveObject {
          contents {
            type {
              repr
            }
          }
        }
      }
    }
  }
}
```
